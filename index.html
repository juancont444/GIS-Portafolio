# ── Mapa de amenaza de incendios sobre
# ── cobertura vegetal · Norte de Santander

import geopandas as gp
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
import contextily as ctx

# Cargar GeoJSON de municipios
capa_municipios = "data/municipios.geojson"
gdf_municipios = gp.read_file(capa_municipios)

# Definir municipios por nivel de alerta
red_alerts_list    = []
orange_alerts_list = []
yellow_alerts_list = [
    'CONVENCION', 'EL CARMEN',
    'OCAÑA', 'TEORAMA'
]

fig, ax = plt.subplots(1, 1, figsize=(10, 10))

# Base: todos los municipios en gris
gdf_municipios.plot(
    ax=ax, color='lightgray',
    edgecolor='black', linewidth=0.5, alpha=0.7
)

# Alerta amarilla
gdf_yellow = gdf_municipios[
    gdf_municipios['MUNICIPIO'].isin(yellow_alerts_list)
]
if not gdf_yellow.empty:
    gdf_yellow.plot(
        ax=ax, color='yellow',
        edgecolor='black', linewidth=0.5
    )

# Mapa base hillshade (Esri)
ctx.add_basemap(
    ax, crs=gdf_municipios.crs,
    source=ctx.providers.Esri.WorldShadedRelief
)

# Leyenda personalizada
ax.legend(handles=[
    mpatches.Patch(color='red',       label='Alerta Roja'),
    mpatches.Patch(color='orange',    label='Alerta Naranja'),
    mpatches.Patch(color='yellow',    label='Alerta Amarilla'),
    mpatches.Patch(color='lightgray', label='Sin alerta'),
], loc=(0.02, 0.05), title='Alertas')

ax.set_title(
    "MAPA DE PRONÓSTICO DE AMENAZA DE INCENDIOS\n"
    "SOBRE LA COBERTURA VEGETAL · NORTE DE SANTANDER",
    fontdict={'fontsize':9,'fontweight':'bold','color':'darkblue'}
)
plt.grid(True, linestyle='-', alpha=0.7)
plt.show()

# ── Mapa de alertas hidrológicas por cuencas
# ── Norte de Santander

capa_cuencas = "data/cuencas.geojson"
gdf_cuencas  = gp.read_file(capa_cuencas)

print(gdf_cuencas['NOM_CUENCA'].unique())

red_alerts_list    = []
orange_alerts_list = []
yellow_alerts_list = [
    'RIO PAMPLONITA',
    'RIO ZULIA'
]

fig, ax = plt.subplots(1, 1, figsize=(10, 10))

gdf_cuencas.plot(
    ax=ax, color='lightgray',
    edgecolor='black', linewidth=0.5, alpha=0.7
)

gdf_yellow = gdf_cuencas[
    gdf_cuencas['NOM_CUENCA'].isin(yellow_alerts_list)
]
if not gdf_yellow.empty:
    gdf_yellow.plot(
        ax=ax, color='yellow',
        edgecolor='black', linewidth=0.5
    )

# Etiquetas de cuencas
for idx, row in gdf_cuencas.iterrows():
    plt.annotate(
        text=row['NOM_CUENCA'],
        xy=row.geometry.centroid.coords[0],
        xytext=(0, 5), textcoords='offset points',
        ha='center', fontsize=5, color='black',
        bbox=dict(boxstyle='square,pad=0',
                  fc='white', ec='none', alpha=0.7)
    )

ctx.add_basemap(
    ax, crs=gdf_cuencas.crs,
    source=ctx.providers.Esri.WorldShadedRelief
)

ax.set_title(
    "MAPA DE ALERTAS HIDROLÓGICAS · NORTE DE SANTANDER",
    fontdict={'fontsize':9,'fontweight':'bold','color':'darkblue'}
)
plt.grid(True, linestyle='-', alpha=0.7)
plt.show()

Boletín Hidroclimatológico — Norte de Santander

Script Python desarrollado para la generación automatizada de mapas de pronóstico de amenazas hidroclimatológicas. Produce tres tipos de mapas a partir de shapefiles municipales y de cuencas hidrográficas.

Mapas generados

1. Amenaza de incendios sobre cobertura vegetal — municipios clasificados por nivel de alerta con mapa base de relieve sombreado Esri.

2. Amenaza de deslizamientos de tierra — riesgo de remoción en masa, especialmente en la zona de cordillera (Chítaga, Toledo).

3. Alertas hidrológicas por cuencas — polígonos de cuencas hidrográficas: Río Pamplonita y Río Zulia en alerta amarilla.

Estructura del repositorio

index.html — página principal del portfolio
data/municipios.geojson — polígonos de municipios (EPSG:4326)
data/cuencas.geojson — polígonos de cuencas (EPSG:4326)
boletin_hidroclimatico.py — script original de Google Colab
